def digit_sum(n):
    s = 0
    while n:   # while n is 'truthy' for an integer that means not 0
        s = s + (n % 10)   # the sum is the sum + the remainder of dividing the incoming number (n) by 10  157 % 10 = 7
        n = n / 10         # n = the integer of n / 10   int(15.7) = 15
    return s

print digit_sum(157)
